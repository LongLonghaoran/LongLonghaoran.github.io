pre_sale_orders = PreSaleOrder.where("created_at > ?", Time.new(2021,1,21)).where.not("total_quantity is not null and total_price is not null")
pre_sale_orders.each do |pre_sale_order|
  begin
    order_datas = OrderCenterService.all_by_order_number(pre_sale_order.order_number)
    if order_datas.present?
      order_data = order_datas[0]
      pre_sale_order.update(
          pre_sale_order_type: :crawl, purchase_date: order_data['purchaseDate'], 
          skus: order_data['skus'], marketplace: order_data['marketplace'],
          total_quantity: order_data['itemQty'], total_price: order_data['totalPrice']
      )
    else
        raise CscError::OrderNotFoundError.new("Order Not Found")
    end
  rescue CscError::OrderNotFoundError => exception
    puts "订单未找到:#{pre_sale_order.id}"
  end
end
