3.解决回调地狱
    为了解决嵌套造成的回调地狱，在ECMAScript 6 中新增了一个api,叫做 Promise
使用Promise:
// 在ECMAScript 6中新增了一个API Promise
// Promise是一个构造函数，浏览器也可以用
var fs = require('fs');
// 1.创建Promise容器，将异步任务放在其中
// 容器一旦创建就开始执行里面的代码
// 2.Promise容器构造函数有2个参数，通过管他们叫做 resolved成功 rejected失败
// 容器默认状态是pending
var promise = new Promise(function(resolved, rejected){
    fs.readFile('./data/a.txt', 'utf8', function(err, data){
        if(err){
            // 失败了，承诺容器中的任务失败了
            // 把容器的Pending状态改为失败


            //这里的reject调用就相当于调用了then方法的第二个参数
            rejected(err);
        }else{
            // 承诺容器中的任务成功了
            // 把容器的Pending状态改为成功 Resolved


            // 这里的resolved方法调用，实际上就相当于是then方法传递的第一个function参数的调用
            resolved(data);
        }
    });
});
// p1就是那个承诺
// 当 p1 成功了，然后就是(then)做指定的操作
// then(function(data){}) , then方法 接收的function就相当于容器中的resolved函数,参数也就是
// resolved方法的参数 data
promise.then(function(data){
    // 之前resolved的时候传递的是data,所以这里接收的也是data
    console.log(data)
}, function(err){
    console.log('读取文件失败了', err);
});

4.通过链式调用解决多级回调的问题
var fs = require('fs');
var p1 = new Promise(function(resolved, rejected){
    fs.readFile('./data/a.txt', 'utf8', function(err, data){
        if(err){
            rejected(err);
        }else{
            resolved(data);
        }
    });
});
var p2 = new Promise(function(resolved, rejected){
    fs.readFile('./data/b.txt', 'utf8', function(err, data){
        if(err){
            rejected(err);
        }else{
            resolved(data);
        }
    });
});
var p3= new Promise(function(resolved, rejected){
    fs.readFile('./data/c.txt', 'utf8', function(err, data){
        if(err){
            rejected(err);
        }else{
            resolved(data);
        }
    });
});
p1.then(function(data){
    // 之前resolved的时候传递的是data,所以这里接收的也是data
    console.log(data);

    // 当return一个promise对象的时候，后续的then方法的第一个参数会作为p2的resolved方法
    // 并且，后续的then方法的第一个function参数的参数，也就是对应着p2的resolved中的参数
    return p2;
}, function(err){
    console.log('读取文件失败了', err);
}).then(function(data){
    // 这里的data即表示p2中的resolved方法接收的参数,也就是p2中读取的文件数据
    console.log(data);
    // 执行完之后，继续返回p3
    return p3;
}).then(function(data){
    console.log(data);
});


// 总之,then的方法的第一个方法参数，总是会在该promise成功之后才会调用
// 所以代码的执行顺序总是会和then的书写顺序是一致的

5.例子，封装readFile
var fs = require('fs');
function pReadFile(path){
    var p1 = new Promise(function(resolved, rejected){
        fs.readFile(path, 'utf8', function(err, data){
            if(err){
                rejected(err);
            }else{
                resolved(data);
            }
        });
    });
    return p1;
}
pReadFile('./data/a.txt')
    .then(function(data){
        console.log(data);
        return pReadFile('./data/b.txt');
    })
    .then(function(data){
        console.log(data);
        return pReadFile('./data/c.txt');
    }).then(function(data){
        console.log(data);
    });


6.Promise应用场景: 级联的下拉框，根据前面一个输入框(或者一个下拉框)的值，来决定后面一个下拉框中的选项
    比如 users 和 jobs
    自己来实现:
        1.数据存储采用mongodb / 或者 使用 json文件(使用监听json文件)
        2.users表结构 {name: String, jobs: [job的id]}
        3.jobs表结构 {name: String}
        4.采用模板引擎渲染页面
        5.采用promise确保第二个下拉框的渲染在第一个输入框/下拉框 的数据获取到之后再执行
    思路: 其实只有2层嵌套的话，直接将第二个下拉框的加载嵌套在第一个函数中也不太影响可读性，可以试想一下
            ，假如在第二个下拉框选中之后，还要继续获取第三个、第四个甚至第五个下拉框的值，这样子再去嵌套的
            话，你会逐渐明白promise的重要性
    提示: 还有一种方法,jquery早就提供了promise类似的链式调用 $.get().then().then()....


  pGet('http://127.0.0.1:3000/users/4')

  function pGet(url, callback) {
    return new Promise(function (resolve, reject) {
      var oReq = new XMLHttpRequest()
      // 当请求加载成功之后要调用指定的函数
      oReq.onload = function () {
        // 我现在需要得到这里的 oReq.responseText
        callback && callback(JSON.parse(oReq.responseText))
        resolve(JSON.parse(oReq.responseText))
      }
      oReq.onerror = function (err) {
        reject(err)
      }
      oReq.open("get", url, true)
      oReq.send()
    })
  }




  // 这个 get 是 callback 方式的 API
  // 可以使用 Promise 来解决这个问题
  function get(url, callback) {
    var oReq = new XMLHttpRequest()
    // 当请求加载成功之后要调用指定的函数
    oReq.onload = function () {
      // 我现在需要得到这里的 oReq.responseText
      callback(oReq.responseText)
    }
    oReq.open("get", url, true)
    oReq.send()
  }
