require 'openssl'
require "base64"

hash  = OpenSSL::HMAC.digest('sha256', "secret", "Message")
puts Base64.encode64(hash)


    require 'openssl'
    require "base64"
    secret = "SECc6d62f28691573317e1207aa4a37802c39f2a39592f19dcb39b31d5c8bd33772"
    timestamp = Time.now.to_i.to_s+"000"
    str = timestamp+"\n"+secret
    digest = OpenSSL::Digest::SHA256.new
    hmac = OpenSSL::HMAC.new(secret, digest)
    hmac << str
    mac = hmac.digest
    sign = Base64.strict_encode64(mac)
    URI.encode(sign)
