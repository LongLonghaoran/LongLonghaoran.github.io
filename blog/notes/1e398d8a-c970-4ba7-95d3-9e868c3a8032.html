1.闭包:
    一个函数有权访问另一个函数作用域中的变量

2.闭包案例:
    页面上有4个li,要求点击每个li时，输出它的索引号
    
错误示范:
    var lis = document.querySelectorAll('li');
    for(var i =0; i< lis.length; i++){
        lis[i].onclick = function(){
            console.log(i);
        }    
    }    
// 由于onclick事件的回调函数是异步事件，等到点击时才会执行，而那时i已经等于4了，所以无论点击哪个li，都是输出4

解答:
    var lis = document.querySelectorAll('li');
    for(var i =0; i< lis.length; i++){
        (function(j){
            lis[i].onclick = function(){
                console.log(j);
            }   
        })(i) 
    }  
//但是，这种做法在性能上的损失非常大，每次循环都创建了一个独立的函数作用域，如果函数不调用的话（也就是一直不进行点击）那么该函数作用域将一直保留并占用内存空间

4.封装深拷贝方法
    
        // 深拷贝拷贝多层, 每一级别的数据都会拷贝.
        var obj = {
            id: 1,
            name: 'andy',
            msg: {
                age: 18
            },
            color: ['pink', 'red']
        };
        var o = {};
        // 封装函数
        function deepCopy(newobj, oldobj) {
            for (var k in oldobj) {
                // 判断我们的属性值属于那种数据类型
                // 1. 获取属性值  oldobj[k]
                var item = oldobj[k];
                // 2. 判断这个值是否是数组
                if (item instanceof Array) {
                    newobj[k] = [];
                    deepCopy(newobj[k], item)
                } else if (item instanceof Object) {
                    // 3. 判断这个值是否是对象
                    newobj[k] = {};
                    deepCopy(newobj[k], item)
                } else {
                    // 4. 属于简单数据类型
                    newobj[k] = item;
                }


            }
        }
        deepCopy(o, obj);
        console.log(o);
