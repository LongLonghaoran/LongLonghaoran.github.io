1.将路由分离出来单独定义路由
    如果将路由全部定义在 app.js 中，那么app.js将会过于肥胖,最好是将路由部分 分离 出去
    基本思路:
        1.建立 router.js
            使用 module.exports = function(app){}直接返回一个方法，之后将直接传递 app作为参数进来
            
module.exports = function(app){
    app.get('/', function(req, res){
        ...
    });
}

        2.在app.js中引入，并且调用，调用的时候将 app 作为参数传入

var router = require('./router');
var app = express();
router(app);

2.express提供了一种更好的方式,使用路由router来分离路由代码
// router.js
//创建一个路由容器
var router = express.Router();
//返回该容器
module.exports = router;

// app.js
var router = require('./router')
//使用该路由容器
app.use(router)

3.引入 body-parser用于处理post数据
var bodyParser = require('body-parser')
// 配置模板引擎和 body-parser 一定要在 app.use(router) 挂载路由之前
// parse application/x-www-form-urlencoded
app.use(bodyParser.urlencoded({ extended: false }))
// parse application/json
app.use(bodyParser.json())


4.Array数组的find方法
    该方法类似于ruby的select方法，可以从数组中选择出符合条件的元素
    var arr = [{a: 1, b:2}, {a: 'a', b: '3}]
    arr.find(function(item){ return item.a ===1}) 
    需要注意的是，回调函数中务必使用 return 来返回 boolean值
