Benchmark.bmbm(10) do |x|
  x.report("sql") do 
    @from_date = DateTime.new(2019,11,1)
    @to_date = DateTime.new(2019,11,30)
    @ticket_responses = TicketResponse.left_joins(:ticket).
    where(created_at: @from_date..@to_date).
    select(
      "tickets.kind ticket_kind", "tickets.offline ticket_offline", "tickets.staff_id ticket_staff_id",
      "tickets.complexity ticket_complexity", "tickets.call_ticket ticket_call_ticket", 
      "ticket_responses.kind kind", "tickets.safety_issue ticket_safety_issue", 
      "ticket_responses.follow_up", "ticket_responses.id", "ticket_responses.staff_id staff_id"
    )
  end

  x.report("ar object") do 
    @from_date = DateTime.new(2019,11,1)
    @to_date = DateTime.new(2019,11,30)
    @ticket_responses = TicketResponse.connection.execute(TicketResponse.left_joins(:ticket).
    where(created_at: @from_date..@to_date).
    select(
      "tickets.kind ticket_kind", "tickets.offline ticket_offline", "tickets.staff_id ticket_staff_id",
      "tickets.complexity ticket_complexity", "tickets.call_ticket ticket_call_ticket", 
      "ticket_responses.kind kind", "tickets.safety_issue ticket_safety_issue", 
      "ticket_responses.follow_up", "ticket_responses.id", "ticket_responses.staff_id staff_id"
    ).to_sql)
  end
end

Struct.new("Student", :name, :age)
students = []
100_000.times do |i|
  students << Struct::Student.new("eddy#{rand(1_000)}", 20+rand(10))
end
students_arr = students.collect{|s| [s.name, s.age]}
# 测试数组和struct对象速度对比
Benchmark.bmbm do |x|
  x.report("Struct") do
    students.select{|s| s.age > 25 }
  end
  x.report("Array") do
    students_arr.select{|s| s[1] > 25}
  end
end

@from_date, @to_date = [DateTime.new(2019,11,1), DateTime.new(2019,11,15)]
result = Benchmark.bmbm(5) do |x|
  x.report("hash array") do
    ticket_responses = TicketResponse.connection.execute(TicketResponse.left_joins(:ticket).
      where(created_at: @from_date..@to_date).
      select(
        "tickets.kind ticket_kind", "tickets.offline ticket_offline", "tickets.staff_id ticket_staff_id",
        "tickets.complexity ticket_complexity", "tickets.call_ticket ticket_call_ticket", 
        "ticket_responses.kind kind", "tickets.safety_issue ticket_safety_issue", 
        "ticket_responses.follow_up", "ticket_responses.id", "ticket_responses.staff_id staff_id"
      ).to_sql)
    ticket_responses.select{|tr| Ticket.kinds.invert[tr[5]] == 'artificial_reply'}
  end

  x.report("ar obj") do
    ticket_responses = TicketResponse.left_joins(:ticket).
      where(created_at: @from_date..@to_date).
      select(
        "tickets.kind ticket_kind", "tickets.offline ticket_offline", "tickets.staff_id ticket_staff_id",
        "tickets.complexity ticket_complexity", "tickets.call_ticket ticket_call_ticket", 
        "ticket_responses.kind kind", "tickets.safety_issue ticket_safety_issue", 
        "ticket_responses.follow_up", "ticket_responses.id", "ticket_responses.staff_id staff_id"
      )
    ticket_responses.select{|tr| Ticket.kinds.invert[tr.kind] == 'artificial_reply'}
  end
end
