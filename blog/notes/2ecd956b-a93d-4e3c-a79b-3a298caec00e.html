导出指定bu issue_category的ticket信息给deyang
笔记本：Anker

# 一、flashlight
# 导出条件：
# 1、BU=Appliances、Dept=EN
# 2、按照时间顺序取 2000 条
# 3、issue关联 SKU 属于以下 PN：T1421,T1420,T1422,T1411,T1424,T1423,T1425,T1426
# 通过 issue 查关联 ticket
# 导出字段：
# ticket_id,ticket_number,subject,message_content,message_simple_text
CSV.open(Rails.root.join("ticket_flashlight.csv"), "wb") do |csv|
csv << %w(ticket_id ticket_number subject message_content message_simple_text)
TicketMessage.joins(:ticket=>[:issue_categories, :ticket_ext]).
where(tickets: {pis_bu_id: 59, department_id: 1}).
where(issues: {pn: %w(T1421 T1420 T1422 T1411 T1424 T1423 T1425 T1426)}).
group(:ticket_id).
select("tickets.id ticket_id", "tickets.ticket_number", "ticket_exts.subject", "message_content_id").
order("ticket_messages.created_at desc").
limit(2000).
each do |ticket_message|
csv << [ticket_message.ticket_id, ticket_message.ticket_number, ticket_message.subject, ticket_message.message_content, ticket_message.ml_text]
end
end
# 二、G 系列扫地机
# 导出条件：
# 1、BU=Appliances、Dept=EN
# 2、时间范围：2020/08/01~2021/01/26
# 3、issue关联 SKU 属于以下 PN：T2150, T2250, T2251
# 通过 issue 查关联 ticket
# 导出字段：
# ticket_id,ticket_number,subject,message_content,message_simple_text
CSV.open(Rails.root.join("G_系统扫地机.csv"), "wb") do |csv|
csv << %w(ticket_id ticket_number subject message_content message_simple_text)
tms = TicketMessage.joins(:ticket=>[:issue_categories, :ticket_ext]).
where(tickets: {pis_bu_id: 59, department_id: 1, created_at: Time.new(2020,8,1)..Time.new(2021,1,26)}).
where(issues: {pn: %w(T2150 T2250 T2251)}).
group(:ticket_id).
select("tickets.id ticket_id", "tickets.ticket_number", "ticket_exts.subject",
"substring_index(group_concat(ticket_messages.message_content_id order by ticket_messages.id asc), ',', 1) message_content_id"
).each do |ticket_message|
csv << [ticket_message.ticket_id, ticket_message.ticket_number, ticket_message.subject, ticket_message.message_content, ticket_message.ml_text]
end
end

# EN市场，Appliances -Cleaning Robot，最近6个月，issue 为 “return to charging base immediately after turning on” 和 issue为 “Not return to charging base” 的数据。
# 时间：2020/9/1-2021/3/4
# dept：EN
# BU category：Appliances-Cleaning Robot
# issue category：return to charging base immediately after turning on、Not return to charging base（按 issue category 分成 2 个文件导出）
# 导出字段：ticket_id, ticket_number, subject,message_content,message_simple_text


params1 = {
product_category_pis_bu_id_or_product_category_pis_category_id_or_product_category_pis_subcategory_id_in: 134,
issue_category_id_child_in: [11071],
created_at_gteq: '2020-09-01',
created_at_lteq: '2021-03-04'
}
issues1 = Issue.ransack(params1).result
CSV.open(Rails.root.join("ticket_infos_1.csv"), "wb") do |csv|

TicketMessage.joins(:ticket=>[:issues, :ticket_ext]).
where(created_at: begin_at..end_at).
where(tickets: {department_id: 1}).
where(issues: {id: issues1.map(&:id)}).
group(:ticket_id).
select("tickets.id ticket_id", "tickets.ticket_number", "ticket_exts.subject", "message_content_id").
order("ticket_messages.created_at desc").each do |ticket_message|

end
