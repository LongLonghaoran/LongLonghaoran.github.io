class MyPromise {
  constructor(executor){
    this.status = 'pending';
    this.value = undefined;
    this.fulfieldCallbacks = [];
    let resolve = (v)=>{
      this.status = 'success';
      this.value = v;
      this.fulfieldCallbacks.forEach((c)=>c());
    }
    executor(resolve);
  }

  then(onFuilfield){
    let promise2 = new MyPromise((resolve)=>{
      if(this.status === 'success'){
        resolvePromise(onFuilfield(this.value), promise2, resolve);
      }else if (this.status === 'pending'){
        this.fulfieldCallbacks.push(
          ()=>{
            // console.log('promise2------',promise2)
            let x = onFuilfield(this.value);
            resolvePromise(x, promise2, resolve);
          }
        )
      }
    })
    return promise2;
  }
}

// 用于下一个then方法调用resolve时能够拿到当前返回的x
function resolvePromise(x, promise2, resolve){
  if(x===promise2){
    throw '不能回调自己'
  }
  if(x instanceof MyPromise){
    // 如果then方法的success函数的返回结果仍然是一个promise
    // 那么我们就需要获取到这个promise的成功回调的resolve值
    // 并且需要保证该当前then方法返回的promise在调用then方法时，resolve函数
    // 接收的值就是该promise resolve的值
    x.then((resolvedValue2)=>{
      resolvePromise(resolvedValue2, promise2, resolve);
    })

    // let then = x.then;
    // then.call(x, y => {
    //   // resolve的结果依旧是promise 那就继续解析
    //   resolvePromise(promise2, y, resolve);
    // })
  }else{
    resolve(x)
  }
}

new MyPromise(
  (resolve, reject)=>{
    setTimeout(
      ()=>{
        console.log('2秒过去了');
        resolve(2);
      }, 2000
    )
  }
).then(
  (fulfieldValue)=>{
    console.log('上一步执行结果是', fulfieldValue);
    return new MyPromise(
      (resolve)=>{
        setTimeout(
          ()=>{
            console.log('又是2秒过去了..');
            resolve(22);
          }, 2000
        )
      }
    );
  }
).then(
  (fulfieldValue)=>{
    console.log('这是最后一步了,上一步执行结果是', fulfieldValue);
    return new MyPromise(
      (resolve)=>{
        setTimeout(
          ()=>{
            console.log('又是2秒过去了..');
            resolve(33);
          }, 2000
        )
      }
    )
  }
).then(
  (fulfieldValue)=>{
    console.log('再加一步,上一步执行结果是', fulfieldValue);
  }
