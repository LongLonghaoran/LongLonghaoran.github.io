CSV.open(Rails.application.root.join('csc_reject_reason.csv'), "wb") do |csv|

  file = File.open('/home/deploy/csc_rejected_ticket.xlsx')
  spreadsheet = Roo::Spreadsheet.open(file)
  sheet_names = spreadsheet.sheets
  sheet_names.each do |sheet_name|
    sheet = spreadsheet.sheet(sheet_name)
    rows = sheet.parse('ticket_number': /\ATicket number\z/i)
    ticket_numbers = rows.collect{|r| r[:ticket_number]}.compact
    tickets = Ticket.where(ticket_number: ticket_numbers)
    csv << [sheet_name]
    tickets.each do |ticket|
      csv << [ticket.ticket_number, ticket.ai_replied, ticket.do_not_call_ai_reason, ticket.sensitive_words.collect{|s| s.keywords}.join(' | ') ]
    end
  end
end
