     <script>
//         食物就是一个对象，有 宽、高、背景色、横纵坐标
        (function(){
            var elements = []; //用于保存每个小方块食物
            function Food(x,y,width,height,color){
                this.x = x||0;
                this.y = y||0;
                this.width = width ||20;
                this.height = height || 20;
                this.color = color ||'green';
            };
//            为原型添加初始化方法（作用：在页面上显示食物）
            Food.prototype.init = function(map){
                remove();
                var div = document.createElement("div");
                map.appendChild(div);
//                横纵坐标先随机产生
                this.x = parseInt(Math.random()*(map.offsetWidth/this.width))*this.width;
                this.y = parseInt(Math.random()*(map.offsetHeight/this.height))*this.height;
//                this.style = {
//                    width:this.width+"px",
//                    height: this.height+"px",
//                    backgroundColor: this.color,
//                    position: 'absolute',
//                    left: this.x+'px',
//                    top: this.y+'px'
//                };
                div.style.width = this.width+"px";
                div.style.height = this.height+"px";
                div.style.backgroundColor = this.color;
                div.style.position = 'absolute';
                div.style.left = this.x+'px';
                div.style.top = this.y+'px';
                elements.push(div);
            };
//            私有函数，用于删除食物
            function remove(){
                for(var i=0;i<elements.length;i++){
                    var ele = elements[i];
//                    先删除元素
                    ele.parentElement.removeChild(ele);
//                    再删除数组中对应的保存的元素,splice用于从 i 处 1个元素
                    elements.splice(i, 1);
                };
            }
            window.Food = Food;
        }());
//        自调用函数---小蛇
        (function(){
            var elements = [];
            function Snake(width, height, direction){
                this.width = width || 20;
                this.height = height || 20;
                this.direction = direction || 'right';
                this.body = [
                    {x: 3, y: 2, color:'red'},
                    {x: 2, y: 2, color:'orange'},
                    {x: 1, y: 2, color:'orange'},
                ]
            }
//            初始化，创建小蛇
            Snake.prototype.init = function(map){
//                循环遍历
                for(var i=0;i<this.body.length;i++){
                    var obj = this.body[i];
                    var div = document.createElement("div");
                    map.appendChild(div);
                    div.style.position = 'absolute';
                    div.style.width = this.width+'px';
                    div.style.height = this.height+'px';
                    div.style.left = obj.x*this.width + 'px';
                    div.style.top = obj.y*this.height + 'px';
                    div.style.backgroundColor = obj.color;
                    elements.push(div);
                }
            };
            //            蛇的移动，老师的思路,将原先的蛇删除，再新建一个蛇对象，赋予新的body值
            // body内元素变换规律，除了第一个元素以外，之后的每个元素的x y值变为它前面一个元素的x y，
            // 也就是跟着前面的元素走
            Snake.prototype.move = function(food, map){
//               每次走之前删除原先的节点，再来新创建蛇对象
                remove();
                var i = this.body.length-1;
//                先移动身体节点的，每个节点的定位值改变为它前一个节点的值
                for(;i>0;i--){
                    this.body[i].x = this.body[i-1].x;
                    this.body[i].y = this.body[i-1].y;
                }
//                再来第一个节点（蛇头）
//                根据方向，来决定蛇头的left top值
                switch(this.direction){
                    case 'right':
                        this.body[0].x+=1;break;
                    case 'left':
                        this.body[0].x-=1;break;
                    case 'top':
                        this.body[0].y-=1;break;
                    case 'bottom':
                        this.body[0].y+=1;
                };
//                    判断有没有吃到食物
//                    即，小蛇的头的坐标和食物是否一致
                var headX = this.body[0].x*this.width;
                var headY = this.body[0].y*this.height;
                var foodX = food.x;
                var foodY = food.y;
                console.log(headX +'-----'+foodX);
                if(headX==foodX && headY == foodY){
//                    食物重新生成
                    food.init(map);
//                    给蛇的Body属性加一个元素，这个元素依据蛇吃的时候的方向，来决定新增的元素的相对于最后一个元素的x y值
                    last = this.body[this.body.length-1];
                    this.body.push({
                        x: last.x,
                        y: last.y,
                        color: last.color
                    });
                    switch(this.direction){
                        case 'right':
                            this.body[this.body.length-1].x-=1;break;
                        case 'left':
                            this.body[this.body.length-1].x+=1;break;
                        case 'top':
                            this.body[this.body.length-1].y-=1;break;
                        case 'bottom':
                            this.body[this.body.length-1].y+=1;
                    };

                }
            };
//            删除小蛇
            function remove(){
                var i = elements.length - 1;
                for(;i>=0;i--){
                    var ele = elements[i];
                    ele.parentElement.removeChild(ele);
                    elements.splice(i, 1);
                }
            }
            window.Snake = Snake;
        }());

//        游戏对象
        (function(){
            var that = null;
            var timeId = null;
            function Game(map){
                this.food = new Food();
                this.snake = new Snake();
                this.map = map;
            }
            Game.prototype.init = function(){
                that= this;
                this.food.init(this.map);
//                调用自动移动的方法
                this.runSnake(this.food, this.map);
                this.bindKey();
            };

//        添加原型方法，设置小蛇可以自动的跑起来
            Game.prototype.runSnake = function(food, map){
                timeId = setInterval(function(){
                    this.snake.move(that.food, that.map);
                    this.snake.init(that.map);
//                    横坐标最大值
                    var maxX = map.offsetWidth/this.snake.width;
//                    纵坐标最大值
                    var maxY = map.offsetHeight/this.snake.height;
                    var headX = this.snake.body[0].x;
                    var headY = this.snake.body[0].y;
                    if(headX < 0 ||headX >=maxX || headY < 0 ||headY >=maxY){
//                        撞墙了，停止定时器
                        clearInterval(timeId);
                        alert("撞墙了，游戏结束");
                    }
                }.bind(that), 200);
            };

            Game.prototype.bindKey = function(){
//                获取用户的按键，改变小蛇的方向
                document.onkeydown = function(e){
                    switch(e.keyCode){
                        case 38:
//                            限定蛇不能反方向跑
                            this.snake.direction = this.snake.direction != 'bottom' ? 'top' :  'bottom';
                            break;
                        case 40:
                            this.snake.direction = this.snake.direction != 'top' ? 'bottom' : 'top';
                            break;
                        case 37:
                            this.snake.direction = this.snake.direction != 'right' ? 'left' : 'right';
                            break;
                        case 39:
                            this.snake.direction = this.snake.direction != 'left' ?'right':'left';
                    };
                }.bind(that);
            };
            Game.prototype.pause = function(){
                clearInterval(timeId);
            };
            window.Game = Game;
        }());

//        创建食物处理对象,可用于创建食物等
        var map = document.querySelector(".map");
        var game = new Game(map);
        game.init();

//         开始暂停按钮
         document.querySelector("#btn1").onclick = function(){
             game.runSnake(game.food, map);
         };
        document.querySelector("#btn2").onclick = function(){
            game.pause();
        }
     </script>
