1.基础数据导出提供给IAM导入到,所有部门基础数据(id,name),部门与员工的对应关系,bu与员工的对应关系


CSV.open(Rails.root.join("department_info.csv"), "wb") do |csv|
    csv << ["导出基础数据时,注意需要将department的id赋值为IAM中数据权限的code"]
    csv << %w(id name)
    Department.all.each do |department|
        csv << [department.id, department.name]
    end
end
CSV.open(Rails.root.join("staff_department_infos.csv"), "wb") do |csv|
    csv << %w(id name)
    Staff.all.each do |staff|
        staff.departments.each do |department|
            csv << [staff.user_name, department.name]
        end
    end
end
CSV.open(Rails.root.join("staff_bu_infos.csv"), "wb") do |csv|
    csv << %w(id name)
    Staff.joins(:departments, :pis_categories).distinct("staffs.id").each do |staff|
        staff.pis_categories.each do |bu|
            csv << [staff.user_name, bu.name]
        end
    end
end
