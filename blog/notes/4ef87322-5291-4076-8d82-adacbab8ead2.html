1.在模块加载中，相对路径中的 ./ 不能省略

2.在文件操作 比如 fs.readFile 中，相对路径可以省略 ./

3.使用express
    1.npm init -y 初始化项目
    2.npm install -S express 安装express
    3.写码
var express = require('express');
// 创建app
var app = express();
app.get('/', function(req, res){
    // 需要注意:原先的res.end res.write仍然是可以用的
    // res.end('hello world')
    // 但是使用express提供的send方法，它会自动的帮助我们设置content-type等一些原先需要手动设置的选项
    res.send('hello world');
});
app.listen(3000, function(){
    console.log('启动成功');
});

4.配置服务自动重启/刷新，也就是像rails那样，自动重载新的代码

//安装在全局范围内
npm install -g nodemon
    2.它是一个机遇node.js开发的第三方命令行工具
        安装之后，启动项目就使用
        nodemon app.js 来启动即可，这样几遍更改的代码，也无需重启服务

6.静态资源服务
    //表示以public开头的url,其后面的路径将作为服务器上./public路径后的路径来访问
    app.use('/public/', express.static('./public/'));
    //当省略第一个参数的时候，表示url中的根路径即为对应的服务器上的文件路径
    app.use(express.static('./public/'));

7.配置使用art-template
    1.安装
npm install --save art-template
npm install --save express-art-template
    2.使用
var express = require('express');
var path = require('path');
var app = express();
app.use('/public/', express.static('./public/'));

// view engine setup
// express为response提供了一个render方法，默认是不可以使用的，但是如果配置了模板引擎就可以使用了
// res.render(html模板名, {模板数据})
// 第一个参数不能写路径，默认会去项目中的views目录查找该模板文件
// 也就是说express有一个约定，开发人员把所有视图文件都放到views目录中
// 第一个参数也表示，当渲染以该名称为后缀名的文件时就加载 express-art-template模块
// 可以不用使用.art文件，使用html也是可以的，而且使用html文件还可以被编辑器识别
app.engine('html', require('express-art-template'));
// app.set('view', {
//     debug: process.env.NODE_ENV !== 'production'
// });
// set views用于修改默认的渲染视图的目录,下面的view也可以改为相对于项目根目录下的其他目录
app.set('views', path.join(__dirname, 'views'));
// app.set('view engine', 'art');


app.get('/', function(req, res){
    // res.send('page');
    // render之后，express默认都会将views目录作为相对路径的起始路径开始查找
    res.render('404.html');
}).get('/admin', function(req, res){
    // 第一个参数是相对于视图目录的文件路径，第二个参数是art-template的模板中的参数
    res.render('admin/index.html', {
        title: 'admin管理系统'
    })
}).get('/post', function(req, res){
    res.send('post page');
}).listen(3000, function(){
    console.log('running ... ')
});
