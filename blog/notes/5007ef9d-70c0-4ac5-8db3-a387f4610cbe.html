  include TicketStat
  from_date = Time.zone.parse('2021-03-21')
  to_date = Time.zone.parse('2021-03-27')
  data = ticket_stat_data(from_date, to_date, 'business_unit', 'response_created_at', nil, nil)  
  CSV.open(Rails.root.join("ticket_stat_0321-0320.csv"), "wb") do |csv|
    csv << data[0].as_json.keys
    data.each do |hash|
      csv << hash.as_json.values
    end
  end

RAILS_ENV=production nohup bundle exec rails runner export_ticket_stat.rb 2>&1 &
