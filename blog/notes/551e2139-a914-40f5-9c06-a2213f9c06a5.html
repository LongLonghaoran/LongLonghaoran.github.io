Promise
笔记本：再学前端

new Promise(
    (resolve, reject)=>{
        setTimeout(()=>{console.log("大约2秒过去了");resolve(2);},2000)
    }    
).then(
    (result)=>{
        return new Promise((resolve, reject)=>{
            console.log('上一步resolve结果是', result);
            resolve(8);
        });
    }
).then(
    (result)=>{
        console.log('这是倒数第二步的回调方法，上一步的执行结果是', result)
    }
).then(
    (result)=>{
        console.log('上一步并没有显示返回promise,这一步能接收到什么呢?', result);
    }   
)
then方法接收2个函数参数，一个成功回调函数、一个失败回调函数

new Promise(
    (resolve, reject)=>{
        setTimeout(()=>{console.log("大约2秒过去了");resolve(2);},2000)
    }    
).then((result)=>{
    throw '报错了';
}).then(
    (result)=>{
        return new Promise((resolve, reject)=>{
            console.log('上一步resolve结果是', result);
            resolve(8);
        });
    }
).catch((result)=>{
    console.log('捕获到了异常');
}).then(

).then(
