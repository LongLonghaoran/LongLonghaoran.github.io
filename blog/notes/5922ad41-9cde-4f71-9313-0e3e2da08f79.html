2.初步实现apache功能
    要求:
        1.要求将 e:/www作为根目录
        2.url中的路径将作为对应的服务器上的文件路径来访问对应的文件
        3.当访问根路径 /时，返回 www/index.html页面
        4.访问不存在的文件时，返回404提示
// 来监听e:/www目录
var http = require('http');
var fs = require('fs');
var server = http.createServer();
var wwwDir = 'e:/www';
server.on('request', function(req, res){
    var url = req.url;
    var filePath = `${wwwDir}/index.html`;
    if(url !== '/'){
        filePath = `${wwwDir}${url}`;
    }
    fs.readFile(filePath, function(err, data){
        if(err){
            return res.end('404 Not Found');
        }else{
            res.end(data);
        }
    });
});
server.listen(3000, function(){
   console.log('running...')
});
