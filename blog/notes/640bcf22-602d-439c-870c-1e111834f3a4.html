rows = []
error_lines = []
File.open(Rails.root.join('response_template.csv')).each do |line|
  begin
    CSV.parse(line) do |row|
      rows << row
    end
  rescue CSV::MalformedCSVError => er
    puts "Error!:"
    error_lines << line
  end
end
rows[1..-1].each do |row|
  response_template = ResponseTemplate.find(row[1])
  response_template.update(
    title: row[4],
    content: row[7]
  )
end
