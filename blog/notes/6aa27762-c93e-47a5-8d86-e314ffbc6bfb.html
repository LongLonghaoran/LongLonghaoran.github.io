# 导出 Headphone BU ticket issue
# EN市场，Headphone 下最近1个月的所有邮件。
# 时间：2021/2/15-2021/3/22
# dept：EN
# BU：Headphone
# 导出字段：ticket_id, ticket_number,issue category name
department = Department.find_by_name('en')

bu = PisCategory.business_units.find_by_name 'Headphone'

issues = Issue.joins(:ticket, :issue_category).

where(tickets: {pis_bu_id: bu.id, department_id: department.id}).

where("issues.created_at between ? and ?", Time.new(2021,2,15), Time.new(2021,3,23))

select("tickets.id ticket_id, tickets.ticket_number ticket_number, issue_categories.name issue_category_name")


CSV.open(Rails.root.join('ticket_issues.csv'), 'wb') do |csv|

csv << %w(ticket_id ticket_number issue_category_name)

issues.each do |issue|

csv << [issue.ticket_id, issue.ticket_number, issue.issue_category_name]

end
end
