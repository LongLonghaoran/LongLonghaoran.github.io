1机器已经安装好了python3.9
2.yum install libpg-devel python3-devel
3.安装postgresql
yum install postgresql-server
postgresql-setup initdb
systemctl start postgresql
systemctl status postgresql
passwd postgres 更改postgres默认用户的密码为postgres
useradd pulp
passwd pulp更改pulp的密码为pulp
su postgres
psql进入postgres控制台
CREATE USER pulp WITH PASSWORD 'pulp';
CREATE DATABASE pulp;
GRANT ALL PRIVILEGES ON DATABASE pulp to pulp;
vim /var/lib/pgsql/data/pg_hba.conf 调整认证方式，让其可以使用pulp连接上数据库
[图片/附件]
4.python3 -m venv pulpvenv
5.source pulpvenv/bin/activate
6.pip install pulpcore
7.cd /root/pulpenv/lib/python3.9/site-packages/pulpcore
8.配置项目
vim app/settings.py
调整数据库连接信息
[图片/附件]
CONTENT_ORIGIN:
[图片/附件]
创建DB_ENCRYPTION_KEY指定的密钥文件
openssl rand -base64 32 > /etc/pulp/certs/database_fields.symmetric.key
创建MEIDA_ROOT指定的目录
mkdir -p /var/lib/pulp/media
chmod 777 /var/lib/pulp/media
chmod 777 /var/lib/pulp
[图片/附件]
创建tmp目录
mkdir -p /var/lib/pulp/tmp
chmod 777  /var/lib/pulp/tmp
执行数据库迁移:
pulpcore-manager migrate --noinput
重置admin密码
pulpcore-manager reset-admin-password --password admin
编译静态资源,pulp界面css
pulpcore-manager collectstatic --noinput
9.启动项目:
apiserver:
pulpcore-manager runserver 0.0.0.0:24817
pulp content:
pulp-content(在pulpenv下是可以这么直接启动，因为可执行文件已经放在/root/pulpenv/bin/pulp-content)
启动worker: pulpcore-worker
安装pulp_deb
pip install pulp_deb
pulpcore-manager migrate --noinput
安装pulp_congtainer
pip install pulp_container
