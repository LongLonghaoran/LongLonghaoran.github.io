导出指定bu tickets
笔记本：Anker


# 导出条件：
# BU =headphone,charger
# 时间：2021/01/01 to 2021/01/31

# 导出字段：
# bu_name,ticket_id,ticket_number,issue_category_name,description
bu_ids = PisCategory.where(name: %w(headphone charger)).pluck(:id)
depts = Department.where(language: 'EN').pluck(:id)
CSV.open(Rails.root.join("tickets_info.csv"), "wb") do |csv|
  csv << "bu_name,ticket_id,ticket_number,issue_category_name,description".split(",")
  Ticket.connection.execute(Ticket.where(pis_bu_id: bu_ids).where(created_at: DateTime.new(2021,1,1)..DateTime.new(2021,1,31)).
    left_joins(:issues => [:issue_category]).
    left_joins(:business_unit).
    where("issues.id is not null").
    where(department_id: depts).
    select("pis_categories.name", "tickets.id", "tickets.ticket_number", "issue_categories.name issue_category_name", "issues.description").to_sql).
    each do |ticket|
    csv << ticket
  end
end
