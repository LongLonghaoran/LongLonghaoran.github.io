Feedback出现formated_feedback_date为空时的情况
笔记本：Anker

Tim原版:

select * from feedbacks where formated_feedback_date is null  order by id  desc limit 10000

update feedbacks set formated_feedback_date = '2020-05-05', updated_at = NOW() where  feedback_date = '2020-05-05' and created_at > '2020-05-01';

update voc_messages set message_date = '2020-05-05', updated_at = NOW() where messageable_id in (select id from feedbacks where formated_feedback_date = '2020-05-05' and created_at > '2020-05-01') and messageable_type = 'Feedback';

更改版:

select *
from feedbacks
where 
feedback_date between '2021-01-10' and '2021-01-16'
    AND created_at > '2021-01-09'
    and formated_feedback_date is null


UPDATE
	feedbacks
SET
	formated_feedback_date = feedback_date,
	updated_at = NOW()
WHERE
	feedback_date between '2021-01-10' and '2021-01-16'
    AND created_at > '2021-01-09'
    and formated_feedback_date is null;


UPDATE
	voc_messages
SET
	message_date = '2021-01-10',
	updated_at = NOW()
WHERE
	messageable_id in(
		SELECT
			id FROM feedbacks
		WHERE
			feedback_date between '2021-01-10' and '2021-01-16'
            AND created_at > '2021-01-09'
            and formated_feedback_date is null;)
	AND messageable_type = 'Feedback';
