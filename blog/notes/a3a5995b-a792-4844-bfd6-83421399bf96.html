file = File.open("/home/deploy/不抓取NR的Listing清单.xlsx")
spreadsheet = Roo::Spreadsheet.open(file)
sheet = spreadsheet.sheet(2)
rows = sheet.parse(sku: /\Asku\z/i, marketplace: /\Amarketplace\z/i, asin: /\Aasin\z/i, seller: /\Aseller\z/i, action: /\Aaction\z/i)

# 导出当前状态
CSV.open("/home/deploy/selling状态", "wb") do |csv|
  rows.each do |row|
    selling = Selling.find_by(row.slice(:sku, :marketplace, :asin, :seller))
    if selling.present?
      csv << selling.attributes.values 
    end
  end
end
# 导出不存在的数据并更改状态
CSV.open("/home/deploy/不存在的selling数据", "wb") do |csv|
  rows.each do |row|
    selling = Selling.find_by(row.slice(:sku, :marketplace, :asin, :seller))
    if selling.blank?
      csv << row.values
    else
      selling.update(enable: false)
    end
  end
end
