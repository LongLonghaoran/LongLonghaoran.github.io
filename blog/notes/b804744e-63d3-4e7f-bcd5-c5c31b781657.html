# 导出目前 CSC 用户在打标管理、分信管理、回信管理、信件转移、check QC 的 filter 数量
# system_settings = SystemSetting.where(name: "filters")
# type:
#   {0: tag, 1: assign, 2: reply, 3: transfer, 4: qc}
CSV.open(Rails.root.join("filters_count.csv"), "wb") do |csv|
    csv << %w(name tag assign reply transfer check_qc)
  Staff.joins(:system_settings).distinct("staffs.id").where(is_valid: true, is_lock: false).each do |staff|
    tag_filters_count = 0
    assign_filters_count = 0
    reply_filters_count = 0
    transfer_filters_count = 0
    check_qc_filters_count = 0
    system_setting = staff.system_settings.where(name: "filters").first
    next if system_setting.blank?
    filters_json = JSON.parse(system_setting.content)
    filters_json.each do |config|
      case config['type']
      when 0
        tag_filters_count += 1
      when 1
        assign_filters_count += 1
      when 2
        reply_filters_count += 1
      when 3
        transfer_filters_count += 1
      when 4
        check_qc_filters_count += 1
      end
    end
    csv << [staff.user_name, tag_filters_count, assign_filters_count, reply_filters_count, transfer_filters_count, check_qc_filters_count]
  end
end
