    4.less语法:
        1.变量 variable
            
@charset "UTF-8";
/* 这样的注释在css当中也可以使用,所以在编译的过程当中也生成在css文件*/
// 但是less总还支持双斜杠注释，这样的注释将不会编译在css文件中
/*变量*/
//必须以@为前缀开始 :是等于的意思 必须以;分号结束
//不能以数字开头, 不能包含特殊符号,大小写敏感
@mainColor: #e92322;
@pink: pink;
@className: box;
div {
  background-color: @mainColor;
}
a:hover {
  color: @mainColor;
}
//当需要和其他字符进行拼接的时候使用 @{变量名}
.@{className} {
  color: @mainColor;
}
//某些时候需要加上''引号声明
.hotpink {
  background-color: 'hot@{pink}';
}
//mixin混入

        2.Minxin混入
            1.将通过类定义的样式，以函数调用的形式来书写，从而达到直接使用该样式的效果
@charset "utf-8";
//混入,分为两种
//组合样式的例子
.w50 {
  width: 50%;
}
.f_l {
  float: left;
}
.f_r {
  float: right;
}
//类混入
.w50andf_l{
  .w50();
  .f_l();
}
//函数混入
//定义函数,这样子我们定义的基础样式将不会导入到css文件中
.w50-2(){
  width: 50%;
}
.f_l-2(){
  float: left;
}
.f_r-2(){
  float: right;
}
.w50andf_l-2{
  .w50();
  .f_l();
}
//带参数的函数
.f(@direction){
  float: @direction;
}
.w50andf-l-3{
  .w50();
  //调用的时候必须传参
  .f(left);
}
//带参数默认值的参数
.f2(@direction: left){
  float: @direction;
}
/*还可以将兼容性代码也写进来*/
.borderRadius(@width: 100px){
  border-radius: @width;
  -webkit-border-radius: @width;
  -moz-border-radius: @width;
  -o-border-radius: @width;
  -ms-border-radius: @width;
}
.w50andf4 {
  .w50();
  .f2();
  .borderRadius(30px);
}
//注意点:由于less并不会去检查属性值的正确性，所以就算这里传入一个不符合float属性的值（比如red #333)也是可以的，所以需要自行注意
        3.嵌套
            
@charset "UTF-8";
.wjs_app {
  display: block;
  //在其中的可以书写其子代元素的css,这些样式相当于加上了父元素的前缀
  img {
    display: none;
  }
  // 父元素自身伪类选择不能直接写，因为那样会转化为 .wjs_app :hover 中间有个空格，相当于后代选择器
  //可以使用 & 来指代父元素
  &:hover {
    img {
      display: block;
    }
  }
  //其余的选择器 像 > 直接子代选择器，这些没关系
  > div {
    display: block;
  }
}

4.导入
    为了满足“高内聚，低耦合”的原则，我们可以将不同模块的less分别书写到不同文件，然后在Main.less中引入,less中使用@import来进行引入
    
@charset "UTF-8";
@import "variables";
@import "mixins";
@import "topBar";
a {
  color: @mainColor;
}
