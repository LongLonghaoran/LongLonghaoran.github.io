将输入中所有的单词(以空格或者换行间隔的单词)按每行一个打印出来:
#include <stdio.h>
#define IN 1
#define OUT 0
int main(){
// long nc = 0;
// while(getchar()!=EOF){
// nc++;
// }
// printf("count:%d",nc);
// int c = 0;
// int n = 0;
// while((c=getchar())!=EOF){
// if (c=='\n'){
// n++;
// }
// }
// printf("%d",n);
int nl = 0, nw = 0, nc = 0;
int state = OUT;
int c;
int arr[100][100];
int i =0 ,j = 0;
int printString();
printf("%d",arr[1][1]);
while((c=getchar())!=EOF){
++nc;
if(c=='\n'){
nl++;
}
if(c==' '||c=='\n'||c=='\t'){
// 离开单词了,进入到下一行
i++;
j=0;
state = OUT;
}else if(state == OUT){
// 进入到单词了,开始记录单词
arr[i][j]=c;
// 索引移到下一个字符
j++;
state = IN;
nw++;
}else{
// 仍然在单词中,开始记录单词
arr[i][j]=c;
// 索引移到下一个字符
j++;
}
}
printf("%d %d %d\n",nl,nw,nc);
printString(arr);
return 0;
}

int printString(int arr[100][100]){
printf("output with one word one line:\n");
// 用于标识该行没有输入字符
int end = 0;
for(int i =0 ;i<100;i++){
if(end==1){
break;
}
for(int j=0;j<100;j++){
if(j==0 && arr[i][j] == 0){
end = 1;
}
putchar(arr[i][j]);
}
printf("\n");
}
return 0;
}
main函数之所以习惯返回一个数值，比如无错误时返回0，是因为main作为程序的入口同时也作为程序的结束的位置，需要给执行环境返回一个它的执行结果，一般返回0表示无错误。

数组变量存储的是第一个元素的地址，所以在把数组变量作为形参传递给函数的时候，实参取到的也是第一个元素的地址，所以通过它对数组元素进行操作，将会改变原数组
#include <stdio.h>

int getline1(char arr[10]){
for(int i = 0;i<10;i++){
arr[i] = 'a';
}
return 0;
}

int getline1();

int main(){
char arr[10];
// 函数将会直接在原数组上赋值
getline1(arr);
for(int i = 0;i<10;i++){
putchar(arr[i]);
}
return 0;
}
