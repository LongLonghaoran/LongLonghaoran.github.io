给官网导出 anker\eufy amazon review 根据官网提供的在售 SKU 名单 时间范围：2020/03/01-2021/03/05 查询 sku 对应在 amazon_us/amazon_uk/amazon_uk2/amazon_ca/amazon_us_eufy 的 review 数据： Anker 产品对应店铺： amazon_us/amazon_uk/amazon_uk2/amazon_ca Eufy 产品对应店铺：amazon_uk/amazon_uk2/amazon_u
笔记本：Anker
  # 导出字段： market/asin/create time/review_comment/review_title/current_star/nick_name

spreadsheet = Roo::Spreadsheet.open('/home/deploy/apps/CSC/current/YoptoReviewTemplate.xlsx')
begin_at = Time.new(2020,3,1)
end_at = Time.new(2021,3,5)
sheet_anker = spreadsheet.sheet(5)
rows_anker = sheet_anker.parse('sku': /\AVariant SKU\z/).collect{|r| r[:sku]}
reviews_anker = Review.where(sku: rows_anker, marketplace: %w(amazon_us amazon_uk amazon_uk2 amazon_ca)).
  joins(:voc_messages).
  where(created_at: begin_at..end_at, current_star: [4,5]).
  group("reviews.id").
  select("reviews.sku", "reviews.marketplace", "reviews.asin", "reviews.created_at", "reviews.nick_name", "substring_index(group_concat(title order by voc_messages.id desc), ',', 1) title", "substring_index(group_concat(body order by voc_messages.id desc), ',', 1) body", "substring_index(group_concat(star order by voc_messages.id desc), ',', 1) star")
CSV.open(Rails.root.join('reviews_anker.csv'), "wb") do |csv|
  csv << %w(marketplace asin created_at title body star nick_name)
  reviews_anker.each do |review|
    csv << [review.sku, review.marketplace, review.asin, review.created_at, review.body, review.title, review.star, review.nick_name]
  end  
end
sheet_eufy = spreadsheet.sheet(6)
rows_eufy = sheet_eufy.parse('sku': /\AVariant SKU\z/).collect{|r| r[:sku]}
reviews_eufy = Review.where(sku: rows_eufy, marketplace: %w(amazon_uk amazon_uk2 amazon_us_eufy)).
  joins(:voc_messages).
  where(created_at: begin_at..end_at, current_star: [4,5]).
  group("reviews.id").
  select("reviews.sku", "reviews.marketplace", "reviews.asin", "reviews.created_at", "reviews.nick_name", "substring_index(group_concat(title order by voc_messages.id desc), ',', 1) title", "substring_index(group_concat(body order by voc_messages.id desc), ',', 1) body", "substring_index(group_concat(star order by voc_messages.id desc), ',', 1) star")
CSV.open(Rails.root.join('reviews_eufy.csv'), "wb") do |csv|
  csv << %w(sku marketplace asin created_at title body star nick_name)
  reviews_eufy.each do |review|
    csv << [review.sku, review.marketplace, review.asin, review.created_at, review.body, review.title, review.star, review.nick_name]
  end  
end
