笔记本：Anker

# 导出系统内所有被标记为Technical fans的客户
# 导出字段（保留列名）：Customer email,Customer name,Ticket#,Market,Dept,BU,Brand,Skus,Comment

CSV.open(Rails.root.join("techinical_fans.csv"), "wb") do |csv|
  csv << %w(email name ticket_number marketplace department bu brand skus comment)
  Ticket.joins(:customer_profile, :user).includes(:customer_profile, :user, :brand).
    where(customer_profiles: {technical_fans: true}).
    each do |ticket|
    puts "ticket:#{ticket.ticket_number}"
    csv << [
      ticket.customer_profile.email,
      ticket.user.name,
      ticket.ticket_number,
      ticket.marketplace,
      ticket.department.name,
      ticket.business_unit.name,
      ticket.brand.name,
      ticket.customer_profile.skus,
      ticket.customer_profile.comment
    ]
  end
end
