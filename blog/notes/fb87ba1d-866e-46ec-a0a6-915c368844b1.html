7.沙箱
    概念：就是一个环境，用来隔绝外部环境
    对应到js里面就是一个 立即执行函数
    由于函数执行完成之后函数中的数据、变量就都销毁了，所以如果我们有一些js语句是不希望
    影响到其他的代码的话，就写一个立即执行函数，把我们要写操作都放在立即执行函数中执行
    var str = '外部字符串';
    (function(){
        var str = '内部字符串';
        console.log(str);
    }());
    console.log(str);
    但是，需要注意，立即执行函数内部还是访问到外部变量的,比如上面的代码改一下:
        var str = '外部字符串';
        (function(){
            console.log(str);
        }());
        console.log(str);

10.浅拷贝 深拷贝
    浅拷贝:
        把一个对象的属性全部赋值到另一个对象
        var obj1 = {
            name: 'bmw'
        }
        var obj2 = {}
        for( var key in  obj1){
            obj2[key] = obj1[key]
        }
    深拷贝:
        先来深拷贝的做法:
            var o1 = {
                name:1,
                hobbies: [1,2,3],some: {a:1,b:2}
            };
            var o2 = {}
            function extend(obj1, obj2){
                for( var key in  obj1){
                    if(Object.prototype.toString.call(obj1[key]) == "[object Array]"){
                        obj2[key] = [];
                        extend(obj1[key], obj2[key]);
                    }else if(Object.prototype.toString.call(obj1[key]) == "[object Object]"){
                        obj2[key] = {};
                        extend(obj1[key], obj2[key]);
                    }else{
                        obj2[key] = obj1[key]
                    }
                }
            }
            extend(o1, o2);
            这么做的原因:
                首先我们拷贝的目的，仅仅是希望将对象中的值都拷贝过来供我们使用，而不去影响原先的
                对象，如果是浅拷贝的话，直接将每个键值对赋值给新的对象的话，那么
                当对象中有对象类型或者数组类型的数据的话，它会直接将对象/数组的引用地址
                赋值给新的对象的对应的属性，这样如果我们通过拷贝后的对象去改变这些引用类型的
                数据的话，也会影响原先的对象

    function forDom(node, fn){
        if(node.children.length != 0){
            fn(node);

                forDom(node.children[i], fn);
            }
        }else{

        }
    }
