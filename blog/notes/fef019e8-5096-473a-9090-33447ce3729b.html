1.通过http这个核心模块可以启动服务，并且监听http请求
// 在node中专门提供了一个核心模块: http
// http的职责就是创建编写服务器的
// 加载核心模块http
var http = require('http')
// 返回一个server实例
var server = http.createServer()
// 处理请求,非常简单,就像js中监听事件，直接来监听request请求事件，然后通过回调函数来处理即可
server.on('request', function(){
    // 当客户端请求过来，就会自动触发服务器的request请求事件,并且触发回调函数
    console.log('收到客户端的请求了');
})
// 绑定端口号,启动服务
server.listen(3000, function(){
    console.log('服务启动了');
})
2.接收请求，并且返回响应,可以通过请求事件的回调的两个参数来处理，回调函数有2个参数，request和response分别用来处理请求和响应
var http = require('http')
// 返回一个server实例
var server = http.createServer()
server.on('request', function(request, response){
    // 回调的两个参数 request response
    // request 请求对象，可以用来获取客户端的一些请求信息，比如请求路径
    // response可以用来给客户端发送响应信息
    console.log('收到客户端的请求了');
    console.log(request.url);
    // 返回响应,write可以使用多次，但是最后一定要使用end来结束响应,否则客户端会一致等待
    response.write('hello')
    response.end()
})
server.listen(3000, function(){
    console.log('服务启动了');
})

4.可以通过end方法同时返回响应内容，需要注意，响应的内容只能是二进制数据或者字符串，如果需要返回
    对象 数组这些数据，需要进行序列化 JSON.stringify才可以
server.on('request', function(req, res){
    console.log('收到请求了，请求路径是' + req.url );
    // res.write('hello')
    // res.write('world')
    // 上面的方式比较麻烦，如果只是范湖文本来调试的话，可以直接使用end加上响应内容作为参数来返回
    // res.end('hello nodejs')
    // 根据请求路径发送不同的响应结果
    // 1.获取请求路径
    var url = req.url
    // 2.判断路径并处理
    if(url == "/"){
        res.end('index page')
    }else if(url == '/login'){
        res.end('login page')
    }else if(url == '/products'){
        // 返回数组
        var products = [
            {name: '苹果'},
            {name: '华为'},
            {name: '三星'},
            {name: '小米'}
        ]
        // 但是响应内容只能是字符串或者二进制数据,直接响应对象 数组 布尔值 数字都是不行的
        // 通过JSON.stringfy 反序列化(序列化 JSON.parse)
        res.end(JSON.stringify(products))
    }else{
        res.end('路径不存在')
    }
})

                require('./b.js')
            如果在require的文件，还有require的话，那么也会继续执行第二层require的文件

7.node中没有全局作用域，只有模块级作用域,如果需要交流数据，那么需要 进行导出exports
    也就是说通过require包含的模块之间，并不能够互相访问其变量
    如果需要访问的话需要通过exports进行导出,被导出的对象，将可以被 进行require的文件进行读取
    exports就是一个各个模块都能读取的一个公用的对象，默认是{} 空的,被require的模块中给exports这个对
    象赋予新的属性，然后谁require了这个模块，那么谁就可以得到模块内部的exports接口对象
    1.在被导入的模块中使用 exports.变量名 = 值 进行导出
exports.foo = 'bbb'
    2.在接收模块的文件中使用 一个变量接收导入的 模块
        
var res = require('./b')
console.log(res.foo);
// console.log(foo)

8.获取请求来源的端口号
    以往我们只知道服务器上服务的端口号，其实在发送响应的时候，也是需要知道请求来源的ip地址和端口号        的，在浏览器发送一个请求给服务器之后，客户机也开启了一个端口来与服务端通信
    可以通过request.socket.remotePort 来获取请求的端口号

10,解决中文乱码的问题
    上面通过 end(内容) 直接响应的话，如果内容为中文，那么将会有乱码
    在服务端发送的数据默认是utf8编码,只不过我们响应的时候，需要显示指定一下编码告知浏览器,浏览器如果
    没有收到来自服务端指定的编码的话，那么将按照系统默认的编码去解析（如系统的语言是中文，那么默认
    是gbk的编码）
    通过设置响应头来设置编码,通过 res.setHeader 设置 Content-Type, text/plain表示纯文本，charset即表       示字符编码，内容类型有很多 text/html text/xml image/png 等等等等
res.setHeader("Content-Type", "text/plain; charset=utf-8")

11.响应Html
    1.首先设置content-type为 text/html; charset=utf-8
    2.返回html字符串
var http = require('http')
var server = http.createServer()
server.on('request', function(req, res){
    console.log('收到请求了，请求路径是' + req.url );
    var url = req.url
    // 判断路径并处理
    if(url == '/html'){
        res.setHeader("Content-Type", "text/html; charset=utf-8")
        res.end('<h1>返回个Html给你</h1>')
    }else{
        res.end('路径不存在')
    }
})
server.listen(3000, function(){
    console.log('服务启动了');
})
12.发送文件中的数据以及Content-Type内容类型
    实现服务器功能，以让nodejs启动的服务能够真的渲染服务器上的文件给客户端，也就是我们平时使用的
    服务器功能了，比如直接渲染目录下某个html文件的内容，因为我们不可能手写整个html文件字符串去响       应.....
var http = require('http')
var fs = require('fs')
var server = http.createServer()
server.on('request', function(req, res){
    var url = req.url
    // 读取目录下的文件
    arr = url.split('/')
    var fileName = './resources/' + arr[arr.length-1]
    // 响应给客户端
    fs.readFile(fileName, function(err, data){
        if(err){
            console.log("读取失败:", err)
            res.setHeader("Content-Type", "*/*; charset=utf-8")
            res.end("读取失败:")
        }else {
            console.log("读取成功:", data)
            //这里我直接设置响应所有类型,这样图片 html 纯文本 css js都可以正常获取了
            res.setHeader("Content-Type", "*/*; charset=utf-8")
            res.end(data)
        }


    })
})
server.listen(3000, function(){
    console.log('server is running')
})
